## –¢–µ–æ—Ä–µ—Ç–∏—á–Ω–∞ —á–∞—Å—Ç–∏–Ω–∞

### –©–æ —Ç–∞–∫–µ –≤–≤–µ–¥–µ–Ω–Ω—è —Ç–∞ –≤–∏–≤–µ–¥–µ–Ω–Ω—è –¥–∞–Ω–∏—Ö?
–í–≤–µ–¥–µ–Ω–Ω—è (input) ‚Äì —Ü–µ –æ—Ç—Ä–∏–º–∞–Ω–Ω—è –¥–∞–Ω–∏—Ö –≤—ñ–¥ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á–∞, –∞ –≤–∏–≤–µ–¥–µ–Ω–Ω—è (output) ‚Äì —Ü–µ –≤–∏–≤—ñ–¥ —ñ–Ω—Ñ–æ—Ä–º–∞—Ü—ñ—ó –Ω–∞ –µ–∫—Ä–∞–Ω.
–£ Python –¥–ª—è —Ü—å–æ–≥–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—Ç—å—Å—è —Ñ—É–Ω–∫—Ü—ñ—ó:
- `input()` ‚Äì –æ—Ç—Ä–∏–º—É—î –≤–≤–µ–¥–µ–Ω—ñ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á–µ–º –¥–∞–Ω—ñ —É –≤–∏–≥–ª—è–¥—ñ —Ä—è–¥–∫–∞.
- `print()` ‚Äì –≤–∏–≤–æ–¥–∏—Ç—å –¥–∞–Ω—ñ –Ω–∞ –µ–∫—Ä–∞–Ω.

### –û—Å–Ω–æ–≤–Ω—ñ –ø—Ä–∏–Ω—Ü–∏–ø–∏ –≤–≤–µ–¥–µ–Ω–Ω—è –¥–∞–Ω–∏—Ö
```python

print("–ü—Ä–∏–≤—ñ—Ç,", name, "!")
```

**–ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü—ñ—è –≤–≤–µ–¥–µ–Ω–∏—Ö –¥–∞–Ω–∏—Ö:**

–§—É–Ω–∫—Ü—ñ—è `input()` –∑–∞–≤–∂–¥–∏ –ø–æ–≤–µ—Ä—Ç–∞—î —Ä—è–¥–æ–∫ (`str`). –Ø–∫—â–æ –Ω–µ–æ–±—Ö—ñ–¥–Ω–æ –æ—Ç—Ä–∏–º–∞—Ç–∏ —á–∏—Å–ª–æ, –ø–æ—Ç—Ä—ñ–±–Ω–æ –≤–∏–∫–æ–Ω–∞—Ç–∏ –ø—Ä–∏–≤–µ–¥–µ–Ω–Ω—è —Ç–∏–ø—É:
```python

print("–í–∞—à –≤—ñ–∫:", age, "–í–∞—à –∑—Ä—ñ—Å—Ç:", height)
```

### –û—Å–Ω–æ–≤–Ω—ñ –ø—Ä–∏–Ω—Ü–∏–ø–∏ –≤–∏–≤–µ–¥–µ–Ω–Ω—è –¥–∞–Ω–∏—Ö
–§—É–Ω–∫—Ü—ñ—è `print()` –¥–æ–∑–≤–æ–ª—è—î –≤–∏–≤–æ–¥–∏—Ç–∏ —Ç–µ–∫—Å—Ç —ñ –∑–º—ñ–Ω–Ω—ñ, —Ä–æ–∑–¥—ñ–ª—è—é—á–∏ —ó—Ö –∫–æ–º–∞–º–∏:
```python
print("–ú–µ–Ω–µ –∑–≤–∞—Ç–∏", name, "—ñ –º–µ–Ω—ñ", age, "—Ä–æ–∫—ñ–≤.")
```

–§–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–∏–π –≤–∏–≤—ñ–¥ –∑ `f-—Ä—è–¥–∫–∞–º–∏`:

## –ü—Ä–∞–∫—Ç–∏—á–Ω—ñ –∑–∞–≤–¥–∞–Ω–Ω—è

### –†—ñ–≤–µ–Ω—å 1: –ü–æ—á–∞—Ç–∫–æ–≤–∏–π (3 –±–∞–ª–∏)
1. **–ü—Ä–æ—Å—Ç–∏–π –¥—ñ–∞–ª–æ–≥**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å —Å–≤–æ—î —ñ–º'—è, —ñ –ø—Ä–æ–≥—Ä–∞–º–∞ –≤–∏–≤–æ–¥–∏—Ç—å –ø—Ä–∏–≤—ñ—Ç–∞–Ω–Ω—è.
   ```python
   name = input("–í–≤–µ–¥—ñ—Ç—å –≤–∞—à–µ —ñ–º'—è: ")
   print("–ü—Ä–∏–≤—ñ—Ç,", name, "!")
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –î–æ–¥–∞—Ç–∏ –º–æ–∂–ª–∏–≤—ñ—Å—Ç—å –≤–≤–µ–¥–µ–Ω–Ω—è –º—ñ—Å—Ç–∞ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á–∞.

2. **–ó–∞–ø–∏—Ç –≤—ñ–∫—É**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å –≤—ñ–∫, –ø—Ä–æ–≥—Ä–∞–º–∞ –≤–∏–≤–æ–¥–∏—Ç—å, —Å–∫—ñ–ª—å–∫–∏ —Ä–æ–∫—ñ–≤ –π–æ–º—É –±—É–¥–µ —á–µ—Ä–µ–∑ 5 —Ä–æ–∫—ñ–≤.
   ```python
   age = int(input("–í–≤–µ–¥—ñ—Ç—å –≤–∞—à –≤—ñ–∫: "))
   print("–ß–µ—Ä–µ–∑ 5 —Ä–æ–∫—ñ–≤ –≤–∞–º –±—É–¥–µ", age + 5, "—Ä–æ–∫—ñ–≤.")
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –í–∏—Ä–∞—Ö—É–≤–∞—Ç–∏, —Å–∫—ñ–ª—å–∫–∏ —Ä–æ–∫—ñ–≤ –±—É–ª–æ 3 —Ä–æ–∫–∏ —Ç–æ–º—É.

3. **–ü—Ä–æ—Å—Ç–∏–π –∫–∞–ª—å–∫—É–ª—è—Ç–æ—Ä**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å –¥–≤–∞ —á–∏—Å–ª–∞, –ø—Ä–æ–≥—Ä–∞–º–∞ –≤–∏–≤–æ–¥–∏—Ç—å —ó—Ö–Ω—é —Å—É–º—É.
   ```python
   a = int(input("–í–≤–µ–¥—ñ—Ç—å –ø–µ—Ä—à–µ —á–∏—Å–ª–æ: "))
   b = int(input("–í–≤–µ–¥—ñ—Ç—å –¥—Ä—É–≥–µ —á–∏—Å–ª–æ: "))
   print("–°—É–º–∞ —á–∏—Å–µ–ª:", a + b)
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –î–æ–¥–∞—Ç–∏ –º–æ–∂–ª–∏–≤—ñ—Å—Ç—å –æ–±—á–∏—Å–ª–µ–Ω–Ω—è –¥–æ–±—É—Ç–∫—É.

### –†—ñ–≤–µ–Ω—å 2: –°–µ—Ä–µ–¥–Ω—ñ–π (6 –±–∞–ª—ñ–≤)
1. **–†–æ–∑—Ä–∞—Ö—É–Ω–æ–∫ –≤—ñ–∫—É**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å —Ä—ñ–∫ –Ω–∞—Ä–æ–¥–∂–µ–Ω–Ω—è, –ø—Ä–æ–≥—Ä–∞–º–∞ –≤–∏—Ä–∞—Ö–æ–≤—É—î –π–æ–≥–æ –≤—ñ–∫.
   ```python
   birth_year = int(input("–í–≤–µ–¥—ñ—Ç—å —Ä—ñ–∫ –Ω–∞—Ä–æ–¥–∂–µ–Ω–Ω—è: "))
   age = 2025 - birth_year
   print("–í–∞–º –ø—Ä–∏–±–ª–∏–∑–Ω–æ", age, "—Ä–æ–∫—ñ–≤.")
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –í–∏—Ä–∞—Ö—É–≤–∞—Ç–∏ –≤—ñ–∫ —É –º—ñ—Å—è—Ü—è—Ö.

2. **–§–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–∏–π –≤–∏–≤—ñ–¥**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å —Å–≤–æ—î —ñ–º'—è —Ç–∞ –º—ñ—Å—Ç–æ, –ø—Ä–æ–≥—Ä–∞–º–∞ –≤–∏–≤–æ–¥–∏—Ç—å —ñ–Ω—Ñ–æ—Ä–º–∞—Ü—ñ—é —É —Ñ–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–æ–º—É –≤–∏–≥–ª—è–¥—ñ.
   ```python
   name = input("–í–≤–µ–¥—ñ—Ç—å —ñ–º'—è: ")
   city = input("–í–≤–µ–¥—ñ—Ç—å –º—ñ—Å—Ç–æ: ")
   print(f"{name} –ø—Ä–æ–∂–∏–≤–∞—î –≤ –º—ñ—Å—Ç—ñ {city}.")
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –î–æ–¥–∞—Ç–∏ –º–æ–∂–ª–∏–≤—ñ—Å—Ç—å –≤–≤–µ–¥–µ–Ω–Ω—è —Ö–æ–±—ñ.

3. **–†–æ–∑—Ä–∞—Ö—É–Ω–æ–∫ –ø–ª–æ—â—ñ –ø—Ä—è–º–æ–∫—É—Ç–Ω–∏–∫–∞**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å –¥–æ–≤–∂–∏–Ω—É —Ç–∞ —à–∏—Ä–∏–Ω—É, –ø—Ä–æ–≥—Ä–∞–º–∞ –æ–±—á–∏—Å–ª—é—î –ø–ª–æ—â—É.
   ```python
   length = float(input("–í–≤–µ–¥—ñ—Ç—å –¥–æ–≤–∂–∏–Ω—É: "))
   width = float(input("–í–≤–µ–¥—ñ—Ç—å —à–∏—Ä–∏–Ω—É: "))
   print("–ü–ª–æ—â–∞ –ø—Ä—è–º–æ–∫—É—Ç–Ω–∏–∫–∞:", length * width)
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –û–±—á–∏—Å–ª–∏—Ç–∏ –ø–µ—Ä–∏–º–µ—Ç—Ä –ø—Ä—è–º–æ–∫—É—Ç–Ω–∏–∫–∞.

---

### –†—ñ–≤–µ–Ω—å 3: –î–æ—Å—Ç–∞—Ç–Ω—ñ–π (9 –±–∞–ª—ñ–≤)
1. **–ü–µ—Ä–µ–≤–µ–¥–µ–Ω–Ω—è –≥—Ä–∞–¥—É—Å—ñ–≤**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É –≤ –¶–µ–ª—å—Å—ñ—è—Ö, –ø—Ä–æ–≥—Ä–∞–º–∞ –∫–æ–Ω–≤–µ—Ä—Ç—É—î —É –§–∞—Ä–µ–Ω–≥–µ–π—Ç–∏.
   ```python
   celsius = float(input("–í–≤–µ–¥—ñ—Ç—å —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É —É –¶–µ–ª—å—Å—ñ—è—Ö: "))
   fahrenheit = celsius * 9/5 + 32
   print("–¢–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ —É –§–∞—Ä–µ–Ω–≥–µ–π—Ç–∞—Ö:", fahrenheit)
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –î–æ–¥–∞—Ç–∏ –∫–æ–Ω–≤–µ—Ä—Ç–∞—Ü—ñ—é —É –ö–µ–ª—å–≤—ñ–Ω–∏.

2. **–§—ñ–Ω–∞–Ω—Å–æ–≤–∏–π –∫–∞–ª—å–∫—É–ª—è—Ç–æ—Ä**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å —Å—É–º—É –¥–µ–ø–æ–∑–∏—Ç—É —Ç–∞ –≤—ñ–¥—Å–æ—Ç–∫–æ–≤—É —Å—Ç–∞–≤–∫—É, –ø—Ä–æ–≥—Ä–∞–º–∞ –æ–±—á–∏—Å–ª—é—î –ø—Ä–∏–±—É—Ç–æ–∫ –∑–∞ —Ä—ñ–∫.
   ```python
   deposit = float(input("–í–≤–µ–¥—ñ—Ç—å —Å—É–º—É –¥–µ–ø–æ–∑–∏—Ç—É: "))
   rate = float(input("–í–≤–µ–¥—ñ—Ç—å —Ä—ñ—á–Ω—É –≤—ñ–¥—Å–æ—Ç–∫–æ–≤—É —Å—Ç–∞–≤–∫—É: "))
   profit = deposit * rate / 100
   print("–ü—Ä–∏–±—É—Ç–æ–∫ –∑–∞ —Ä—ñ–∫:", profit)
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –î–æ–¥–∞—Ç–∏ —Ä–æ–∑—Ä–∞—Ö—É–Ω–æ–∫ –∑–∞–≥–∞–ª—å–Ω–æ—ó —Å—É–º–∏ —á–µ—Ä–µ–∑ —Ä—ñ–∫.

3. **–ö–æ–Ω–≤–µ—Ä—Ç–µ—Ä –≤–∞–ª—é—Ç–∏**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å —Å—É–º—É –≤ –≥—Ä–∏–≤–Ω—è—Ö, –ø—Ä–æ–≥—Ä–∞–º–∞ –ø–µ—Ä–µ–≤–æ–¥–∏—Ç—å —ó—ó –≤ –¥–æ–ª–∞—Ä–∏.
   ```python
   uah = float(input("–í–≤–µ–¥—ñ—Ç—å —Å—É–º—É –≤ –≥—Ä–∏–≤–Ω—è—Ö: "))
   usd = uah / 39.5  # –ö—É—Ä—Å –Ω–∞ –º–æ–º–µ–Ω—Ç –Ω–∞–ø–∏—Å–∞–Ω–Ω—è
   print("–°—É–º–∞ —É –¥–æ–ª–∞—Ä–∞—Ö:", round(usd, 2))
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—è:** –î–æ–¥–∞—Ç–∏ –∫–æ–Ω–≤–µ—Ä—Ç–∞—Ü—ñ—é –≤ —î–≤—Ä–æ.

### –†—ñ–≤–µ–Ω—å 4: –í–∏—Å–æ–∫–∏–π (12 –±–∞–ª—ñ–≤)
1. **–ö–∞–ª—å–∫—É–ª—è—Ç–æ—Ä —Ä–æ–∑—Å—Ç—Ä–æ—á–∫–∏**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å —Å—É–º—É –ø–æ–∫—É–ø–∫–∏ —Ç–∞ –∫—ñ–ª—å–∫—ñ—Å—Ç—å –º—ñ—Å—è—Ü—ñ–≤, –ø—Ä–æ–≥—Ä–∞–º–∞ —Ä–æ–∑—Ä–∞—Ö–æ–≤—É—î —â–æ–º—ñ—Å—è—á–Ω–∏–π –ø–ª–∞—Ç—ñ–∂.
   **–ü—Ä–∏–∫–ª–∞–¥ –∫–æ–¥—É:**
   ```python
   # –ó–∞–ø–∏—Ç—É—î–º–æ —Å—É–º—É –ø–æ–∫—É–ø–∫–∏ —Ç–∞ –∫—ñ–ª—å–∫—ñ—Å—Ç—å –º—ñ—Å—è—Ü—ñ–≤
   purchase_amount = float(input("–í–≤–µ–¥—ñ—Ç—å —Å—É–º—É –ø–æ–∫—É–ø–∫–∏: "))
   months = int(input("–í–≤–µ–¥—ñ—Ç—å –∫—ñ–ª—å–∫—ñ—Å—Ç—å –º—ñ—Å—è—Ü—ñ–≤: "))

   # –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞, —â–æ–± –∫—ñ–ª—å–∫—ñ—Å—Ç—å –º—ñ—Å—è—Ü—ñ–≤ –±—É–ª–∞ –±—ñ–ª—å—à–æ—é –∑–∞ 0
   if months > 0:
       monthly_payment = purchase_amount / months
       print("–©–æ–º—ñ—Å—è—á–Ω–∏–π –ø–ª–∞—Ç—ñ–∂ —Å—Ç–∞–Ω–æ–≤–∏—Ç—å:", monthly_payment)
   else:
       print("–ö—ñ–ª—å–∫—ñ—Å—Ç—å –º—ñ—Å—è—Ü—ñ–≤ –ø–æ–≤–∏–Ω–Ω–∞ –±—É—Ç–∏ –±—ñ–ª—å—à–æ—é –∑–∞ –Ω—É–ª—å!")
    ```
    **–í–∞—Ä—ñ–∞—Ü—ñ—ó** –í–∞—Ä—ñ–∞–Ω—Ç –∑ –≤—ñ–¥—Å–æ—Ç–∫–æ–≤–æ—é —Å—Ç–∞–≤–∫–æ—é: –ó–∞–ø–∏—Ç–∞–π—Ç–µ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á–∞ –ø—Ä–æ –≤—ñ–¥—Å–æ—Ç–∫–æ–≤—É —Å—Ç–∞–≤–∫—É —Ç–∞ –≤—Ä–∞—Ö—É–π—Ç–µ —ó—ó –ø—Ä–∏ –æ–±—á–∏—Å–ª–µ–Ω–Ω—ñ —â–æ–º—ñ—Å—è—á–Ω–æ–≥–æ –ø–ª–∞—Ç–µ–∂—É.
    ```python
    purchase_amount = float(input("–í–≤–µ–¥—ñ—Ç—å —Å—É–º—É –ø–æ–∫—É–ø–∫–∏: "))
    months = int(input("–í–≤–µ–¥—ñ—Ç—å –∫—ñ–ª—å–∫—ñ—Å—Ç—å –º—ñ—Å—è—Ü—ñ–≤: "))
    interest_rate = float(input("–í–≤–µ–¥—ñ—Ç—å —Ä—ñ—á–Ω—É –≤—ñ–¥—Å–æ—Ç–∫–æ–≤—É —Å—Ç–∞–≤–∫—É (—É –≤—ñ–¥—Å–æ—Ç–∫–∞—Ö): "))

    if months > 0:
        # –†–æ–∑—Ä–∞—Ö—É–Ω–æ–∫ —â–æ–º—ñ—Å—è—á–Ω–æ–≥–æ –≤—ñ–¥—Å–æ—Ç–∫–æ–≤–æ–≥–æ –ø–ª–∞—Ç–µ–∂—É
        monthly_interest = (purchase_amount * (interest_rate / 100)) / 12
        monthly_payment = (purchase_amount / months) + monthly_interest
        print("–©–æ–º—ñ—Å—è—á–Ω–∏–π –ø–ª–∞—Ç—ñ–∂ —Å—Ç–∞–Ω–æ–≤–∏—Ç—å:", round(monthly_payment, 2))
    else:
        print("–ö—ñ–ª—å–∫—ñ—Å—Ç—å –º—ñ—Å—è—Ü—ñ–≤ –ø–æ–≤–∏–Ω–Ω–∞ –±—É—Ç–∏ –±—ñ–ª—å—à–æ—é –∑–∞ –Ω—É–ª—å!")
    ```
    **–û–∫—Ä—É–≥–ª–µ–Ω–Ω—è**: –û–∫—Ä—É–≥–ª—ñ—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç –¥–æ 2 –¥–µ—Å—è—Ç–∫–æ–≤–∏—Ö –∑–Ω–∞–∫—ñ–≤ –∑–∞ –¥–æ–ø–æ–º–æ–≥–æ—é —Ñ—É–Ω–∫—Ü—ñ—ó round().
2. **–ß–∞—Å–æ–≤–∏–π –∫–æ–Ω–≤–µ—Ä—Ç–µ—Ä**
   - –ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–≤–æ–¥–∏—Ç—å –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å–µ–∫—É–Ω–¥, –ø—Ä–æ–≥—Ä–∞–º–∞ –ø–µ—Ä–µ–≤–æ–¥–∏—Ç—å —É –≥–æ–¥–∏–Ω–∏, —Ö–≤–∏–ª–∏–Ω–∏, —Å–µ–∫—É–Ω–¥–∏.
   ```python
   # –ó–∞–ø–∏—Ç—É—î–º–æ –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å–µ–∫—É–Ω–¥
   total_seconds = int(input("–í–≤–µ–¥—ñ—Ç—å –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å–µ–∫—É–Ω–¥: "))

   # –†–æ–∑—Ä–∞—Ö—É–Ω–æ–∫ –≥–æ–¥–∏–Ω, —Ö–≤–∏–ª–∏–Ω —Ç–∞ —Å–µ–∫—É–Ω–¥
   hours = total_seconds // 3600
   minutes = (total_seconds % 3600) // 60
   seconds = total_seconds % 60

   print(f"{total_seconds} —Å–µ–∫—É–Ω–¥ —Ü–µ {hours} –≥–æ–¥–∏–Ω, {minutes} —Ö–≤–∏–ª–∏–Ω, {seconds} —Å–µ–∫—É–Ω–¥.")
   ```
   **–í–∞—Ä—ñ–∞—Ü—ñ—ó**:
   1. –§–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–∏–π –≤–∏–≤—ñ–¥: –í–∏–≤–µ–¥—ñ—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç —É —Ñ–æ—Ä–º–∞—Ç—ñ –≥–æ–¥:—Ö–≤:—Å—Å
   ```python
   print(f"{hours:02d}:{minutes:02d}:{seconds:02d}")
   ```
   2. –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ –Ω–µ–≥–∞—Ç–∏–≤–Ω–æ–≥–æ –≤–≤–µ–¥–µ–Ω–Ω—è: –î–æ–¥–∞–π—Ç–µ –ø–µ—Ä–µ–≤—ñ—Ä–∫—É, —â–æ–± –æ–±—Ä–æ–±–∏—Ç–∏ –≤–∏–ø–∞–¥–æ–∫, –∫–æ–ª–∏ –≤–≤–µ–¥–µ–Ω–æ –Ω–µ–≥–∞—Ç–∏–≤–Ω–µ —á–∏—Å–ª–æ.

   ```python
   total_seconds = int(input("–í–≤–µ–¥—ñ—Ç—å –∫—ñ–ª—å–∫—ñ—Å—Ç—å —Å–µ–∫—É–Ω–¥: "))
    ```

3. **–†–æ–∑—Ä–∞—Ö—É–Ω–æ–∫ —Å–µ—Ä–µ–¥–Ω—å–æ—ó –æ—Ü—ñ–Ω–∫–∏**
